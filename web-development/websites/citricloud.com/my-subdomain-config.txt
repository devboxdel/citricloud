  # my.citricloud.com - Dashboard subdomain (serves same app as main domain)
  server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name my.citricloud.com;

    # SSL Configuration
    ssl_certificate /etc/letsencrypt/live/citricloud.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/citricloud.com/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Frontend static files (same as main domain)
    root /home/ubuntu/infrastructure/cloud/web-development/websites/citricloud.com/frontend/dist;
    index index.html;

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    # Static assets with caching
    location ~* \.(?:css|js|mjs|map)$ {
      access_log off;
      expires 1y;
      add_header Cache-Control "public, max-age=31536000, immutable" always;
      add_header X-Content-Type-Options "nosniff" always;
      try_files $uri =404;
    }

    location ~* \.(?:png|jpe?g|gif|svg|webp|ico|woff2?|ttf|otf)$ {
      access_log off;
      expires 1y;
      add_header Cache-Control "public, max-age=31536000, immutable" always;
      add_header X-Content-Type-Options "nosniff" always;
      try_files $uri =404;
    }

    # API proxy (CitriCloud backend on port 8000)
    location /api/ {
      proxy_pass http://127.0.0.1:8000/api/;
      proxy_http_version 1.1;
      proxy_set_header Host $host;
      proxy_set_header X-Real-IP $remote_addr;
      proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
      proxy_set_header X-Forwarded-Proto $scheme;
      proxy_set_header Upgrade $http_upgrade;
      proxy_set_header Connection "upgrade";
      proxy_read_timeout 3600s;
    }

    # SPA fallback for React Router
    location = /index.html {
      add_header Cache-Control "no-cache, no-store, must-revalidate, max-age=0" always;
      add_header Pragma "no-cache" always;
      add_header Expires "0" always;
      add_header X-Content-Type-Options "nosniff" always;
      etag off;
    }

    location / {
      try_files $uri $uri/ /index.html;
    }
  }
